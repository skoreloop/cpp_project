cmake_minimum_required(VERSION 3.10)
project(cpp_project)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

include_directories(include)

# 1. Для SFML 3 имена компонентов изменились!
find_package(SFML 3 COMPONENTS Graphics Window System REQUIRED)

set(SOURCES
    main.cpp
    src/Visualizer.cpp
    src/engine/particles.cpp
    src/engine/forces_calculation.cpp
)

add_executable(cpp_project ${SOURCES})

# 2. Линкуем через новые цели SFML::
target_link_libraries(cpp_project SFML::Graphics SFML::Window SFML::System)
